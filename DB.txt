-- 위도,경도 좌표간의 거리 구하기
SELECT 가져올컬럼,

(6371 * acos(cos(radians(lat좌표값)) * cos(radians(slLat)) * cos(radians(slLng) - radians(lng좌표값)) +

sin(radians( lat좌표값)) * sin(radians(slLat))))

AS distance

FROM 대상테이블

HAVING distance <= 거리

ORDER BY distance 

LIMIT 0,1000
